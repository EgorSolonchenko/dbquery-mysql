Example

$query->where('foo', 10); // WHERE foo = 10
$query->where('foo > ?', 10); // WHERE foo > 10
$query->where('foo IS NULL'); // WHERE foo IS NULL
$query->where('foo', array(10, 20)); // WHERE foo IN (10, 20)
$query->where('foo BETWEEN ? AND ?', array(10, 20)); // WHERE foo BETWEEN 10 AND 20
$query->where('bar LIKE %?%', "blue"); // WHERE bar LIKE "%blue%"
$query->where('foo = ? AND bar LIKE %?%', array(10, "blue")); // WHERE foo = 10 AND bar LIKE "%blue%"
$query->where(array('foo'=>10, 'bar'=>"blue")); // WHERE foo = 10 AND bar = "blue"
